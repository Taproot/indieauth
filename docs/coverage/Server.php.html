364        if (!$config['authorizationForm'] instanceof AuthorizationFormInterface) {
365            throw new BadMethodCallException("When provided, \$config['authorizationForm'] must implement Taproot\IndieAuth\Callback\AuthorizationForm.");
366        }
367        $this->authorizationForm = $config['authorizationForm'];
368        trySetLogger($this->authorizationForm, $this->logger);
369    }

372        return $this->tokenStorage;
373    }
374
375    /**
376     * Handle Authorization Endpoint Request
377     * 
378     * This method handles all requests to your authorization endpoint, passing execution off to
379     * other callbacks when necessary. The logical flow can be summarised as follows:
380     * 
381     * * If this request an **auth code exchange for profile information**, validate the request
382     *   and return a response or error response.
383     * * Otherwise, proceed, wrapping all execution in CSRF-protection middleware.
384     * * Validate the request’s indieauth authorization code request parameters, returning an 
385     *   error response if any are missing or invalid.
386     * * Call the authentication callback
387     *     * If the callback returned an instance of ResponseInterface, the user is not currently
388     *       logged in. Return the Response, which will presumably start an authentication flow.
389     *     * Otherwise, the callback returned information about the currently logged-in user. Continue.
390     * * If this request is an authorization form submission, validate the data, store and authorization
391     *   code and return a redirect response to the client redirect_uri with code data. On an error, return
392     *   an appropriate error response.
393     * * Otherwise, fetch the client_id, parse app data if present, validate the `redirect_uri` and present
394     *   the authorization form/consent screen to the user.
395     * * If none of the above apply, try calling the non-indieauth request handler. If it returns a Response,
396     *   return that, otherwise return an error response.
397     * 
398     * This route should NOT be wrapped in additional CSRF-protection, due to the need to handle API 
399     * POST requests from the client. Make sure you call it from a route which is excluded from any
400     * CSRF-protection you might be using. To customise the CSRF protection used internally, refer to the
401     * {@see Server::__construct()} config array documentation for the `csrfMiddleware` key.
402     * 
403     * Most user-facing errors are thrown as instances of {@see IndieAuthException}, which are passed off to
404     * `handleException` to be turned into an instance of `ResponseInterface`. If you want to customise
405     * error handling, one way to do so is to subclass `Server` and override that method.
406     * 
407     * @param ServerRequestInterface $request
408     * @return ResponseInterface
409     */
410    public function handleAuthorizationEndpointRequest(ServerRequestInterface $request): ResponseInterface {
411        $this->logger->info('Handling an IndieAuth Authorization Endpoint request.');
412        
413        // If it’s a profile information request:
414        if (isIndieAuthAuthorizationCodeRedeemingRequest($request)) {
415            $this->logger->info('Handling a request to redeem an authorization code for profile information.');
416            
417            $bodyParams = $request->getParsedBody();
418
419            if (!isset($bodyParams['code'])) {
420                $this->logger->warning('The exchange request was missing the code parameter. Returning an error response.');
421                return new Response(400, [
422                    'Content-Type' => 'application/json',
423                    'Cache-control' => 'no-store',
424                    'Pragma' => 'no-cache'
425                ], json_encode([
426                    'error' => 'invalid_request',
427                    'error_description' => 'The code parameter was missing.'
428                ]));
429            }
430
431            // Attempt to internally exchange the provided auth code for an access token.
432            // We do this before anything else so that the auth code is invalidated as soon as the request starts,
433            // and the resulting access token is revoked if we encounter an error. This ends up providing a simpler
434            // and more flexible interface for TokenStorage implementors.
435            try {
436                // Call the token exchange method, passing in a callback which performs additional validation
437                // on the auth code before it gets exchanged.
438                $tokenData = $this->tokenStorage->exchangeAuthCodeForAccessToken($bodyParams['code'], function (array $authCode) use ($request, $bodyParams) {
439                    // Verify that all required parameters are included.
440                    $requiredParameters = ($this->requirePkce or !empty($authCode['code_challenge'])) ? ['client_id', 'redirect_uri', 'code_verifier'] : ['client_id', 'redirect_uri'];
441                    $missingRequiredParameters = array_filter($requiredParameters, function ($p) use ($bodyParams) {
442                        return !array_key_exists($p, $bodyParams) || empty($bodyParams[$p]);
443                    });
444                    if (!empty($missingRequiredParameters)) {
445                        $this->logger->warning('The exchange request was missing required parameters. Returning an error response.', ['missing' => $missingRequiredParameters]);
446                        throw IndieAuthException::create(IndieAuthException::INVALID_REQUEST, $request);
447                    }
448
449                    // Verify that it was issued for the same client_id and redirect_uri
450                    if ($authCode['client_id'] !== $bodyParams['client_id']
451                        || $authCode['redirect_uri'] !== $bodyParams['redirect_uri']) {
452                        $this->logger->error("The provided client_id and/or redirect_uri did not match those stored in the token.");
453                        throw IndieAuthException::create(IndieAuthException::INVALID_GRANT, $request);
454                    }
455
456                    // If the auth code was requested with no code_challenge, but the exchange request provides a 
457                    // code_verifier, return an error.
458                    if (!empty($bodyParams['code_verifier']) && empty($authCode['code_challenge'])) {
459                        $this->logger->error("A code_verifier was provided when trying to exchange an auth code requested without a code_challenge.");
460                        throw IndieAuthException::create(IndieAuthException::INVALID_GRANT, $request);
461                    }
462
463                    if ($this->requirePkce or !empty($authCode['code_challenge'])) {
464                        // Check that the supplied code_verifier hashes to the stored code_challenge
465                        // TODO: support method = plain as well as S256.
466                        if (!hash_equals($authCode['code_challenge'], generatePKCECodeChallenge($bodyParams['code_verifier']))) {
467                            $this->logger->error("The provided code_verifier did not hash to the stored code_challenge");
468                            throw IndieAuthException::create(IndieAuthException::INVALID_GRANT, $request);
469                        }
470                    }
471
472                    // Check that this token either grants at most the profile scope.
473                    $requestedScopes = array_filter(explode(' ', $authCode['scope'] ?? ''));
474                    if (!empty($requestedScopes) && $requestedScopes != ['profile']) {
475                        $this->logger->error("An exchange request for a token granting scopes other than “profile” was sent to the authorization endpoint.");
476                        throw IndieAuthException::create(IndieAuthException::INVALID_GRANT, $request);
477                    }
478                });
479            } catch (IndieAuthException $e) {
480                // If an exception was thrown, return a corresponding error response.
481                return new Response(400, [
482                    'Content-Type' => 'application/json',
483                    'Cache-control' => 'no-store',
484                    'Pragma' => 'no-cache'
485                ], json_encode([
486                    'error' => $e->getInfo()['error'],
487                    'error_description' => $e->getMessage()
488                ]));
489            }
490
491            if (is_null($tokenData)) {
492                $this->logger->error('Attempting to exchange an auth code for a token resulted in null.', $bodyParams);
493                return new Response(400, [
494                    'Content-Type' => 'application/json',
495                    'Cache-control' => 'no-store',
496                    'Pragma' => 'no-cache'
497                ], json_encode([
498                    'error' => 'invalid_grant',
499                    'error_description' => 'The provided credentials were not valid.'
500                ]));
501            }
502
503            // TODO: return an error if the token doesn’t contain a me key.
504
505            // If everything checked out, return {"me": "https://example.com"} response
506            return new Response(200, [
507                'Content-Type' => 'application/json',
508                'Cache-control' => 'no-store',
509                'Pragma' => 'no-cache'
510            ], json_encode(array_filter($tokenData, function (string $k) {
511                // Prevent codes exchanged at the authorization endpoint from returning any information other than
512                // me and profile.
513                return in_array($k, ['me', 'profile']);
514            }, ARRAY_FILTER_USE_KEY)));
515        }
516
517        // Because the special case above isn’t allowed to be CSRF-protected, we have to do some rather silly
518        // closure gymnastics here to selectively-CSRF-protect requests which do need it.
519        return $this->csrfMiddleware->process($request, new Middleware\ClosureRequestHandler(function (ServerRequestInterface $request) {
520            // Wrap the entire user-facing handler in a try/catch block which catches any exception, converts it
521            // to IndieAuthException if necessary, then passes it to $this->handleException() to be turned into a
522            // response.
523            try {
524                $queryParams = $request->getQueryParams();
525
526                /** @var ResponseInterface|null $clientIdResponse */
527                /** @var string|null $clientIdEffectiveUrl */
528                /** @var array|null $clientIdMf2 */
529                list($clientIdResponse, $clientIdEffectiveUrl, $clientIdMf2) = [null, null, null];
530
531                // If this is an authorization or approval request (allowing POST requests as well to accommodate 
532                // approval requests and custom auth form submission.
533                if (isIndieAuthAuthorizationRequest($request, ['get', 'post'])) {
534                    $this->logger->info('Handling an authorization request', ['method' => $request->getMethod()]);
535
536                    // Validate the Client ID.
537                    // isClientIdentifier is strict about client IDs containing path segments. For the moment we want to
538                    // be a little more lenient about that, so we normalize it to include a path segment before checking.
539                    if (!isset($queryParams['client_id']) || false === filter_var($queryParams['client_id'], FILTER_VALIDATE_URL) || !isClientIdentifier(IndieAuthClient::normalizeMeURL($queryParams['client_id']))) {
540                        $this->logger->warning("The client_id provided in an authorization request was not valid.", $queryParams);
541                        throw IndieAuthException::create(IndieAuthException::INVALID_CLIENT_ID, $request);
542                    }
543
544                    // Validate the redirect URI.
545                    if (!isset($queryParams['redirect_uri']) || false === filter_var($queryParams['redirect_uri'], FILTER_VALIDATE_URL)) {
546                        $this->logger->warning("The redirect_uri provided in an authorization request was not valid.", $queryParams);
547                        throw IndieAuthException::create(IndieAuthException::INVALID_REDIRECT_URI, $request);
548                    }
549
550                    // How most errors are handled depends on whether or not the request has a valid redirect_uri. In
551                    // order to know that, we need to also validate, fetch and parse the client_id.
552                    // If the request lacks a hash, or if the provided hash was invalid, perform the validation.
553                    $currentRequestHash = hashAuthorizationRequestParameters($request, $this->secret, null, null, $this->requirePkce);
554                    if (!isset($queryParams[self::HASH_QUERY_STRING_KEY]) or is_null($currentRequestHash) or !hash_equals($currentRequestHash, $queryParams[self::HASH_QUERY_STRING_KEY])) {
555
556                        // All we need to know at this stage is whether the redirect_uri is valid. If it
557                        // sufficiently matches the client_id, we don’t (yet) need to fetch the client_id.
558                        if (!urlComponentsMatch($queryParams['client_id'], $queryParams['redirect_uri'], [PHP_URL_SCHEME, PHP_URL_HOST, PHP_URL_PORT])) {
559                            // If we do need to fetch the client_id, store the response and effective URL in variables
560                            // we defined earlier, so they’re available to the approval request code path, which additionally
561                            // needs to parse client_id for h-app markup.
562                            try {
563                                list($clientIdResponse, $clientIdEffectiveUrl) = call_user_func($this->httpGetWithEffectiveUrl, IndieAuthClient::normalizeMeURL($queryParams['client_id']));
564                                $clientIdMf2 = Mf2\parse((string) $clientIdResponse->getBody(), $clientIdEffectiveUrl);
565                            } catch (ClientExceptionInterface | RequestExceptionInterface | NetworkExceptionInterface $e) {
566                                $this->logger->error("Caught an HTTP exception while trying to fetch the client_id. Returning an error response.", [
567                                    'client_id' => $queryParams['client_id'],
568                                    'exception' => $e->__toString()
569                                ]);
570
571                                throw IndieAuthException::create(IndieAuthException::HTTP_EXCEPTION_FETCHING_CLIENT_ID, $request, $e);
572                            } catch (Exception $e) {
573                                $this->logger->error("Caught an unknown exception while trying to fetch the client_id. Returning an error response.", [
574                                    'exception' => $e->__toString()
575                                ]);
576
577                                throw IndieAuthException::create(IndieAuthException::INTERNAL_EXCEPTION_FETCHING_CLIENT_ID, $request, $e);
578                            }
579                            
580                            // Search for all link@rel=redirect_uri at the client_id.
581                            $clientIdRedirectUris = [];
582                            if (array_key_exists('redirect_uri', $clientIdMf2['rels'])) {
583                                $clientIdRedirectUris = array_merge($clientIdRedirectUris, $clientIdMf2['rels']['redirect_uri']);
584                            }
585                            
586                            foreach (HeaderParser::parse($clientIdResponse->getHeader('Link')) as $link) {
587                                if (array_key_exists('rel', $link) && mb_strpos(" {$link['rel']} ", " redirect_uri ") !== false) {
588                                    // Strip off the < > which surround the link URL for some reason.
589                                    $clientIdRedirectUris[] = substr($link[0], 1, strlen($link[0]) - 2);
590                                }
591                            }
592
593                            // If the authority of the redirect_uri does not match the client_id, or exactly match one of their redirect URLs, return an error.
594                            if (!in_array($queryParams['redirect_uri'], $clientIdRedirectUris)) {
595                                $this->logger->warning("The provided redirect_uri did not match either the client_id, nor the discovered redirect URIs.", [
596                                    'provided_redirect_uri' => $queryParams['redirect_uri'],
597                                    'provided_client_id' => $queryParams['client_id'],
598                                    'discovered_redirect_uris' => $clientIdRedirectUris
599                                ]);
600
601                                throw IndieAuthException::create(IndieAuthException::INVALID_REDIRECT_URI, $request);
602                            }
603                        }                        
604                    }
605
606                    // From now on, we can assume that redirect_uri is valid. Any IndieAuth-related errors should be
607                    // reported by redirecting to redirect_uri with error parameters.
608
609                    // Validate the state parameter.
610                    if (!isset($queryParams['state']) or !isValidState($queryParams['state'])) {
611                        $this->logger->warning("The state provided in an authorization request was not valid.", $queryParams);
612                        throw IndieAuthException::create(IndieAuthException::INVALID_STATE, $request);
613                    }
614                    // From now on, any redirect error responses should include the state parameter.
615                    // This is handled automatically in `handleException()` and is only noted here
616                    // for reference.
617
618                    // If either PKCE parameter is present, validate both.
619                    if (isset($queryParams['code_challenge']) or isset($queryParams['code_challenge_method'])) {
620                        if (!isset($queryParams['code_challenge']) or !isValidCodeChallenge($queryParams['code_challenge'])) {
621                            $this->logger->warning("The code_challenge provided in an authorization request was not valid.", $queryParams);
622                            throw IndieAuthException::create(IndieAuthException::INVALID_CODE_CHALLENGE, $request);
623                        }
624    
625                        if (!isset($queryParams['code_challenge_method']) or !in_array($queryParams['code_challenge_method'], ['S256', 'plain'])) {
626                            $this->logger->error("The code_challenge_method parameter was missing or invalid.", $queryParams);
627                            throw IndieAuthException::create(IndieAuthException::INVALID_CODE_CHALLENGE, $request);
628                        }
629                    } else {
630                        // If neither PKCE parameter is defined, and PKCE is required, throw an error. Otherwise, proceed.
631                        if ($this->requirePkce) {
632                            $this->logger->warning("PKCE is required, and both code_challenge and code_challenge_method were missing.");
633                            throw IndieAuthException::create(IndieAuthException::INVALID_REQUEST_REDIRECT, $request);
634                        }
635                    }
636
637                    // Validate the scope parameter, if provided.
638                    if (array_key_exists('scope', $queryParams) && !isValidScope($queryParams['scope'])) {
639                        $this->logger->warning("The scope provided in an authorization request was not valid.", $queryParams);
640                        throw IndieAuthException::create(IndieAuthException::INVALID_SCOPE, $request);
641                    }
642
643                    // Normalise the me parameter, if it exists.
644                    if (array_key_exists('me', $queryParams)) {
645                        $queryParams['me'] = IndieAuthClient::normalizeMeURL($queryParams['me']);
646                        // If the me parameter is not a valid profile URL, ignore it.
647                        if (false === $queryParams['me'] || !isProfileUrl($queryParams['me'])) {
648                            $queryParams['me'] = null;
649                        }
650                    }
651
652                    // Build a URL containing the indieauth authorization request parameters, hashing them
653                    // to protect them from being changed.
654                    // Make a hash of the protected indieauth-specific parameters. If PKCE is in use, include 
655                    // the PKCE parameters in the hash. Otherwise, leave them out.
656                    $hash = hashAuthorizationRequestParameters($request, $this->secret, null, null, $this->requirePkce);
657                    // Operate on a copy of $queryParams, otherwise requests will always have a valid hash!
658                    $redirectQueryParams = $queryParams;
659                    $redirectQueryParams[self::HASH_QUERY_STRING_KEY] = $hash;
660                    $authenticationRedirect = $request->getUri()->withQuery(buildQueryString($redirectQueryParams))->__toString();
661                    
662                    // User-facing requests always start by calling the authentication request callback.
663                    $this->logger->info('Calling handle_authentication_request callback');
664                    $authenticationResult = call_user_func($this->handleAuthenticationRequestCallback, $request, $authenticationRedirect, $queryParams['me'] ?? null);
665                    
666                    // If the authentication handler returned a Response, return that as-is.
667                    if ($authenticationResult instanceof ResponseInterface) {
668                        return $authenticationResult;
669                    } elseif (is_array($authenticationResult)) {
670                        // Check the resulting array for errors.
671                        if (!array_key_exists('me', $authenticationResult)) {
672                            $this->logger->error('The handle_authentication_request callback returned an array with no me key.', ['array' => $authenticationResult]);
673                            throw IndieAuthException::create(IndieAuthException::AUTHENTICATION_CALLBACK_MISSING_ME_PARAM, $request);
674                        }
675
676                        // If this is a POST request sent from the authorization (i.e. scope-choosing) form:
677                        if (isAuthorizationApprovalRequest($request)) {
678                            // Authorization approval requests MUST include a hash protecting the sensitive IndieAuth
679                            // authorization request parameters from being changed, e.g. by a malicious script which
680                            // found its way onto the authorization form.
681                            if (!array_key_exists(self::HASH_QUERY_STRING_KEY, $queryParams)) {
682                                $this->logger->warning("An authorization approval request did not have a " . self::HASH_QUERY_STRING_KEY . " parameter.");
683                                throw IndieAuthException::create(IndieAuthException::AUTHORIZATION_APPROVAL_REQUEST_MISSING_HASH, $request);
684                            }
